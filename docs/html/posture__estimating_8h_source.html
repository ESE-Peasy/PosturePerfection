41 /*
42  * @brief A representation of a body part
43  *
44  * Each body part is represented as a `ConnectedJoint` to record the joints
45  * relative position and the connected joints to this joint
46  *
47  * Angles are measured clockwise from Head
48  *
49  * ^HEAD
50  * |
51  * |/ +30 degrees
52  * |
53  * |__ +90 degrees
54  */
55 
56 struct ConnectedJoint {
57  Joint joint;
58  PostProcessing::Coordinate coord;
59  ConnectedJoint * upper_connected_joint;
60  float upper_angle;
61  ConnectedJoint * lower_connected_joint;
62  float lower_angle;
63 }

65 /*
66  * @brief The representation of a human's pose, containing all
67  * the expected `ConnectedJoint`

69 struct Pose {
70  ConnectedJoint head;
71  ConnectedJoint neck;
72  ConnectedJoint shoulder;
73  ConnectedJoint hip;

78 /*
79  * @brief This class handles representations of the user's pose and
80  * calculates any updates to their pose that is required.
81  *
82  * This class stores representations of the users current pose and their
83  * calibrated ideal pose, from `PostProcessing::ProcessedResult` class.
84  *
85  * This class handles:
86  * Updating current user's pose or ideal user's pose.
87  * Calculating the angle between `ConnectedJoint`
88  * Comparing the current user's pose and ideal user's pose.
89  *
90  * General use should just consist of `updateCurrentPoseAndCheckPosture`.
91  *
92  */
93 class PostureEstimator {
94  private:
95  /*
96  * @brief Calculates the angle(in degrees) between two points, clockwise
97  * from the Head.
98  *
99  * @param coord1 Coordinate 1
100  * @param coord2 Coordinate 2
101  *
102  * @returns float
103  */
104  float getLineAngle(
105  PostProcessing::Coordinate coord1,
106  PostProcessing::Coordinate coord2);
107 
108  /*
109  * @brief Converts `PostProcessing::ProcessedResults` to a Pose.
110  *
111  * @param results The `PostProcessing::ProccessedResults` struct
112  * from `PostProcessing::PostProcessor` being run.
113  *
114  * @return `Pose`
115  */
116  Pose createPose(PostProcessing::ProcessedResults results);
117 
118  /*
119  * @brief Updates the user's current pose from `PostProcessing::ProccessedResults`.
120  *
121  * @param results `PostProcessing::ProcessedResults struct containing user's pose data.
122  */
123  void update_current_pose(PostProcessing::ProcessedResults results);
124 
125  /*
126  * @brief Compares user's current pose and ideal pose, returing a pose
127  * object, with the angles of the `ConnectedJoint` representing the change
128  * needed for each one to return current pose to ideal pose.
129  *
130  * @returns Pose
131  */
132  Pose calculatePoseChanges();
133 
134  /*
135  * @brief Checks the current pose_changes_needed and sees if they are still
136  * within the good posture threshold. WARNING Does not calculate posture
137  * changes. Designed to be used in conjunction with calculate
138  * pose_changes
139  */
140  bool checkGoodPosture();
141 
142  /*
143  * @brief calculates current posture and ideal posture difference and decides
144  * if is a good posture based of posture threshold values.
145  * @return true If user's posture is good
146  * @return false If user's posture is bad
147  */
148  bool calculateChangesAndCheckPosture();
149 
150 
151  public:
152  /*
153  * @brief The user's current pose from most recent data provided.
154  */
155  Pose current_pose;
156 
157  /*
158  * @brief The user's ideal pose, expected to be generated
159  * during user calibration.
160  */
161  Pose ideal_pose;
162 
163  /*
164  * @brief A representation of what changes are needed to get
165  * back to ideal pose.
166  */
167  Pose pose_changes;
168 
169  /*
170  * @brief A representation of what absolute value changes for a pose
171  * is acceptably still a good pose.
172  */
173  float pose_change_threshold;
174 
175  /*
176  * @brief Whether the user is in a good pose (true) or bad pose (false).
177  */
178  bool good_pose;
179 
180  /*
181  * @brief Updates the user's ideal pose from `PostProcessing::ProcessingResults`.
182  *
183  * @param results `PostProcessing::ProcessedResults` struct containing user's pose data.
184  */
185  void update_ideal_pose(PostProcessing::ProcessedResults results);
186 
187  /*
188  * @brief Updates the user's current pose from `PostProcessing::ProcessingResults`,
189  * Calculates pose change needed.
190  * Checks if pose change needed is outside threshold.
191  * Returns true or false if user in bad pose.
192  *
193  * @param results `PostProcessing::ProcessingResults` struct containing user's pose data.
194  *
195  * @return true If user's posture is good
196  * @return false If user's posture is bad
197  */
198  bool updateCurrentPoseAndCheckPosture(
199  PostProcessing::ProcessedResults results);
200 }
